function MouseWheelHandler(e, element) {
  var delta = 0;
  if (typeof e === 'number') {
    delta = e;
  } else {
    if (e.deltaX !== 0) {
      delta = e.deltaX;
    } else {
      delta = e.deltaY;
    }
    e.preventDefault();
  }

  element.scrollLeft -= (delta);

}

function SmoothScroll(e) {
  var element = $('#crsl1-items');
  var delta = element[0].scrollLeft - e;
  element.animate({
    scrollLeft: delta
  }, 500);
}

window.onload = function() {
  var crsl1 = {};
  crsl1.e = document.getElementById('crsl1');
  crsl1.items = document.getElementById('crsl1-items');
  crsl1.leftScroll = document.getElementById('1st-left-scroll-button');
  crsl1.rightScroll = document.getElementById('1st-right-scroll-button');

  crsl1.items.addEventListener("mousewheel", handleMouse, false);
  crsl1.items.addEventListener("scroll", scrollEvent);
  crsl1.leftScroll.addEventListener("click", leftScrollClick);
  crsl1.rightScroll.addEventListener("click", rightScrollClick);
 /* crsl1.leftScroll.addEventListener("mousedown", leftScrollClick);
  crsl1.rightScroll.addEventListener("mousedown", rightScrollClick);*/

  setLeftScrollOpacity();
  setRightScrollOpacity();
  
  function handleMouse(e) {
    MouseWheelHandler(e, crsl1.items);
  }

  function leftScrollClick() {
    var smoothScrollDelta = $('#crsl1-items')[0].clientWidth * 0.8;
    SmoothScroll(smoothScrollDelta);
  }

  function rightScrollClick() {
    var smoothScrollDelta = $('#crsl1-items')[0].clientWidth * 0.8;
    SmoothScroll(-smoothScrollDelta);
  }

  function scrollEvent(e) {
    setLeftScrollOpacity();
    setRightScrollOpacity();
  }
  
  function setLeftScrollOpacity() {
    if (isScrolledAllLeft()) {
      crsl1.leftScroll.style.opacity = 0;
    } else {
      crsl1.leftScroll.style.opacity = 1;
    }
  }
  
  function isScrolledAllLeft() {
    if (crsl1.items.scrollLeft === 0) {
      return true;
    } else {
      return false;
    }
  }
  
  function isScrolledAllRight() {
    if (crsl1.items.scrollWidth > crsl1.items.offsetWidth) {
      if (crsl1.items.scrollLeft + crsl1.items.offsetWidth === crsl1.items.scrollWidth) {
        return true;
      } 
    }else {
      return true;
    }
    
    return false;
  }
  
  function setRightScrollOpacity() {
    if (isScrolledAllRight()){
        crsl1.rightScroll.style.opacity = 0;
      } else {
        crsl1.rightScroll.style.opacity = 1;
      }
  }
}

function toDetail(id) {
    console.log(id);
    window.location.href = '/app/sceneReview/review?mvno='+ id;
}
